Quick Start

Deploy your Next.js app in minutes 

# Install YC-OpenNext CLI
npm install -g @yc-opennext/cli

# Build your Next.js application
yc-opennext build

# Deploy to Yandex Cloud
yc-opennext deploy --env production

# View deployment status
yc-opennext status
module "nextjs" {
  source = "yc-opennext/nextjs/yandex"

  app_name     = "my-nextjs-app"
  environment  = "production"
  build_id     = var.build_id

  # Enable features
  enable_isr   = true
  enable_cdn   = true

  # Scaling configuration
  function_memory = 512
  function_timeout = 30
}
// next.config.js
module.exports = {
  output: 'standalone',

  experimental: {
    // Enable ISR with YDB backend
    isrMemoryCacheSize: 0,
  },

  images: {
    domains: ['your-domain.com'],
    formats: ['image/avif', 'image/webp'],
  },
}
